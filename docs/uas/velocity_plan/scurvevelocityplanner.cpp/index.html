class SCurveVelocityPlanner {
public:
    // æ¨¡å‹è¾“å…¥å‚æ•°
    double S;      // æ€»è½¨è¿¹å¼§é•¿ï¼ˆç±³ï¼‰
    double T;      // æ€»é£è¡Œæ—¶é—´ï¼ˆç§’ï¼‰
    double vIn;    // è¿›å…¥é€Ÿåº¦ï¼ˆç±³/ç§’ï¼‰
    double vDes;   // æœŸæœ›é€Ÿåº¦ï¼ˆç±³/ç§’ï¼‰â€”â€”åæ®µå®šé€Ÿ
    double vMin;   // æœ€å°é€Ÿåº¦ï¼ˆç±³/ç§’ï¼‰
    double vMax;   // æœ€å¤§é€Ÿåº¦ï¼ˆç±³/ç§’ï¼‰
    double aMax;   // æœ€å¤§åŠ é€Ÿåº¦ï¼ˆç±³/ç§’Â²ï¼‰
    double jMax;   // æœ€å¤§åŠ åŠ é€Ÿåº¦ï¼ˆç±³/ç§’Â³ï¼‰
    
    // é¢„è®¡ç®—çš„å˜é€Ÿé˜¶æ®µå‚æ•°
    double Tconv;  // å˜é€Ÿé˜¶æ®µæ—¶é•¿ï¼ˆS æ›²çº¿éƒ¨åˆ†ï¼‰
    double Tj;     // ä¸Šå‡/ä¸‹é™é˜¶æ®µæ—¶é•¿ï¼ˆåŠ åŠ é€Ÿåº¦é˜¶æ®µæ—¶é•¿ï¼‰
    double Ta;     // æ¢¯å½¢ S æ›²çº¿ä¸­æ’åŠ é€Ÿé˜¶æ®µæ—¶é•¿ï¼ˆè‹¥ä¸ºä¸‰è§’å½¢åˆ™ä¸º 0ï¼‰
    
    bool isTrapezoidal; // è‹¥é‡‡ç”¨æ¢¯å½¢ S æ›²çº¿åˆ™ä¸º trueï¼Œå¦åˆ™ä¸ºä¸‰è§’å½¢
    
    // æ„é€ å‡½æ•°ï¼šåˆå§‹åŒ–è¾“å…¥å‚æ•°ï¼Œå¹¶è®¡ç®—å˜é€Ÿé˜¶æ®µå‚æ•°
    SCurveVelocityPlanner(double S_, double T_, double vIn_, double vDes_,
                          double vMin_, double vMax_, double aMax_, double jMax_)
    : S(S_), T(T_), vIn(vIn_), vDes(vDes_), vMin(vMin_), vMax(vMax_), aMax(aMax_), jMax(jMax_) {
        computeTransitionTime();
    }
    
    // æ ¹æ® vDes ä¸ vIn çš„å·®å€¼ï¼Œåˆ¤æ–­é‡‡ç”¨æ¢¯å½¢æˆ–ä¸‰è§’å½¢ S æ›²çº¿ï¼Œå¹¶è®¡ç®—ç›¸åº”å‚æ•°
    void computeTransitionTime() {
        double deltaV = vDes - vIn;
        double s = (deltaV >= 0) ? 1.0 : -1.0;
        if (std::fabs(deltaV) < 1e-6) {
            // æ— éœ€å˜é€Ÿ
            Tconv = 0.0;
            Tj = 0.0;
            Ta = 0.0;
            isTrapezoidal = true; // ä»»æ„é€‰æ‹©
        } else {
            if (std::fabs(deltaV) >= (aMax * aMax) / jMax) {
                // æ¢¯å½¢ S æ›²çº¿
                isTrapezoidal = true;
                Tj = aMax / jMax;
                Ta = std::fabs(deltaV) / aMax - Tj;
                Tconv = 2 * Tj + Ta;
            } else {
                // ä¸‰è§’å½¢ S æ›²çº¿
                isTrapezoidal = false;
                Tj = std::sqrt(std::fabs(deltaV) / jMax);
                Tconv = 2 * Tj;
                Ta = 0.0;
            }
        }
    }
    
    // æ ¹æ®å½“å‰æ—¶é—´ t è®¡ç®— S æ›²çº¿çŠ¶æ€è¾“å‡ºï¼ˆä¸è®¡ç®—ç´¯è®¡è·ç¦»ï¼Œç”±å¤–éƒ¨ç§¯åˆ†ï¼‰
    SCurveOutput compute(double t) {
        SCurveOutput out;
        out.time = t;
        double s = (vDes - vIn >= 0) ? 1.0 : -1.0;
        
        // å½“æ— å˜é€Ÿéœ€æ±‚æ—¶
        if (std::fabs(vDes - vIn) < 1e-6) {
            out.velocity = vIn;
            out.acceleration = 0;
            out.jerk = 0;
        }
        else if (t <= Tconv) {
            // å¤„äº S æ›²çº¿å˜é€Ÿé˜¶æ®µ
            if (isTrapezoidal) {
                // æ¢¯å½¢ S æ›²çº¿
                if (t <= Tj) {
                    // ä¸Šå‡é˜¶æ®µï¼šåŠ åŠ é€Ÿé˜¶æ®µ
                    out.jerk = s * jMax;
                    out.acceleration = s * jMax * t;
                    out.velocity = vIn + s * (jMax * t * t) / 2;
                }
                else if (t <= Tj + Ta) {
                    // æ’å®šåŠ é€Ÿé˜¶æ®µ
                    out.jerk = 0;
                    out.acceleration = s * aMax;
                    double vRamp = vIn + s * (jMax * Tj * Tj) / 2; // ä¸Šå‡é˜¶æ®µæœ«é€Ÿåº¦
                    out.velocity = vRamp + s * aMax * (t - Tj);
                }
                else {
                    // ä¸‹é™é˜¶æ®µï¼šåŠ é€Ÿåº¦è¡°å‡é˜¶æ®µ
                    double tau = t - (Tj + Ta);
                    out.jerk = -s * jMax;
                    out.acceleration = s * aMax - s * jMax * tau;
                    double vRamp = vIn + s * (jMax * Tj * Tj) / 2;
                    out.velocity = vRamp + s * aMax * Ta + s * (aMax * tau - (jMax * tau * tau) / 2);
                }
            } else {
                // ä¸‰è§’å½¢ S æ›²çº¿
                if (t <= Tj) {
                    // ä¸Šå‡é˜¶æ®µ
                    out.jerk = s * jMax;
                    out.acceleration = s * jMax * t;
                    out.velocity = vIn + s * (jMax * t * t) / 2;
                } else {
                    // ä¸‹é™é˜¶æ®µ
                    double tau = t - Tj;
                    out.jerk = -s * jMax;
                    out.acceleration = s * jMax * (Tj - tau);
                    double vRamp = vIn + s * (jMax * Tj * Tj) / 2;
                    out.velocity = vRamp + s * (jMax * Tj * tau - (jMax * tau * tau) / 2);
                }
            }
        } else {
            // å®šé€Ÿé˜¶æ®µï¼Œä¿æŒæœŸæœ›é€Ÿåº¦
            out.velocity = vDes;
            out.acceleration = 0;
            out.jerk = 0;
        }
        
        // å¯¹é€Ÿåº¦åšä¸Šä¸‹é™ä¿æŠ¤
        if (out.velocity < vMin) out.velocity = vMin;
        if (out.velocity > vMax) out.velocity = vMax;
        
        return out;
    }
};

// è®¡ç®—æ•´æ®µè½¨è¿¹çš„ S æ›²çº¿æ•°æ®ï¼ˆæŒ‰æ­¥é•¿ dt è®¡ç®—æ¯ä¸ªæ—¶é—´ç‚¹ï¼‰ï¼Œå¹¶è¿”å› vector
std::vector<SCurveOutput> computeTrajectory(SCurveVelocityPlanner &planner, double dt) {
    std::vector<SCurveOutput> trajectory;
    int steps = static_cast<int>(std::ceil(planner.T / dt));
    double cumulativeDistance = 0.0;
    SCurveOutput prevOutput = {0.0, 0.0, planner.vIn, 0.0, 0.0};
    
    for (int i = 0; i <= steps; i++) {
        double t = i * dt;
        if (t > planner.T) t = planner.T; // ç¡®ä¿æœ€åæ—¶åˆ»ä¸è¶…å‡º T
        SCurveOutput out = planner.compute(t);
        // ç”¨ç®€å•æ¢¯å½¢ç§¯åˆ†è®¡ç®—ç´¯è®¡è·ç¦»ï¼ˆå¯ä»¥ä½¿ç”¨å¹³å‡å€¼æé«˜ç²¾åº¦ï¼‰
        if (i == 0) {
            cumulativeDistance = 0.0;
        } else {
            // é‡‡ç”¨å‰åé€Ÿåº¦å¹³å‡è¿‘ä¼¼ç§¯åˆ†
            double avgV = (prevOutput.velocity + out.velocity) / 2.0;
            cumulativeDistance += avgV * dt;
        }
        out.distance = cumulativeDistance;
        trajectory.push_back(out);
        prevOutput = out;
    }
    return trajectory;
}

        return false;

    file << std::fixed << std::setprecision(6);
    
    // å†™å…¥æ¯ä¸ªæ—¶åˆ»æ•°æ®
    for (const auto &pt : trajectory) {

             << pt.jerk << "\n";
    }
    
    file.close();
    return true;
}
